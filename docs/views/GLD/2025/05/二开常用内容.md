select @F_COUNT =  count(F_ID) from TE_INJECT_JS where F_MODULE_CODE = 'GEPS.Contract.Expenditure.QTZCHTModule' and F_PATH = '..\..\..\..\Common\DTJS\QTZCHT_INJECT.js';
if @F_COUNT = 0 THEN
set  @F_ID = @F_ID+1;

END;
